package main

import (
	"fmt"
	"time"
)

func main() {
	// 模拟一些操作，比如循环计数
	for i := 0; i <= 100; i++ {
		// 返回到行首
		fmt.Print("\r")
		// 输出其他内容
		fmt.Printf("Prefix: ")
		// 输出进度
		fmt.Printf("Progress: %d%%", i)
		// 强制刷新输出缓冲区
		fmt.Print("\033[0m") // 重置 ANSI 颜色，防止影响后续输出
		time.Sleep(100 * time.Millisecond)
	}
	fmt.Println("\nDone!") // 完成后换行
}
