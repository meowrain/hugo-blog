Docker安装mysql

2
3
4

$ docker pull mysql:latest
$ cd ~
$ mkdir mysql
$ cd mysql
1

$ docker run -p 3306:3306 --name mysql -v $PWD/conf/my.cnf:/etc/mysql/my.cnf -v $PWD/logs:/logs -v $PWD/data:/mysql_data -e MYSQL_ROOT_PASSWORD=123456 -d mysql
